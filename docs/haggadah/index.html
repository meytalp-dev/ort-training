בָּרוּךְ אַתָּה יְיָ אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם,

בָּרוּךְ אַתָּה יְיָ אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם,
אֲשֶׁר בָּחַר בָּנוּ מִכָּל עָם וְרוֹמְמָנוּ מִכָּל לָשׁוֹן
וְקִדְּשָׁנוּ בְּמִצְוֹתָיו.

וַתִּתֶּן לָנוּ יְיָ אֱלֹהֵינוּ בְּאַהֲבָה
מוֹעֲדִים לְשִׂמְחָה, חַגִּים וּזְמַנִּים לְשָׂשׂוֹן,
אֶת יוֹם חַג הַמַּצּוֹת הַזֶּה,
זְמַן חֵרוּתֵנוּ, מִקְרָא קֹדֶשׁ,
זֵכֶר לִיצִיאַת מִצְרָיִם.
בָּרוּךְ אַתָּה יְיָ אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם,
שֶׁהֶחֱיָנוּ וְקִיְּמָנוּ וְהִגִּיעָנוּ לַזְּמַן הַזֶּה.

🥬 טובלים ירק במי מלח

לוקחים ירק (בדרך כלל כרפס, סלרי או תפוח אדמה) וטובלים אותו במי מלח.

מברכים "בורא פרי האדמה" וטובלים
בָּרוּךְ אַתָּה יְיָ אֱלֹהֵינוּ מֶלֶךְ הָעוֹלָם,
בּוֹרֵא פְּרִי הָאֲדָמָה.

למה מי מלח? מי המלח מסמלים את הדמעות שבני ישראל בכו בזמן השעבוד במצרים. כשאנחנו טובלים את הירק – אנחנו נזכרים בסבל.

סיפור יציאת מצרים

הָא לַחְמָא עַנְיָא דִי אֲכָלוּ אַבְהָתָנָא
בְּאַרְעָא דְמִצְרָיִם.
כָּל דִכְפִין יֵיתֵי וְיֵיכֹל,
כָּל דִצְרִיךְ יֵיתֵי וְיִפְסַח.
הָשַּׁתָּא הָכָא, לְשָׁנָה הַבָּאָה בְּאַרְעָא דְיִשְׂרָאֵל.
הָשַּׁתָּא עַבְדֵי, לְשָׁנָה הַבָּאָה בְּנֵי חוֹרִין.

תרגום: זהו לחם העוני שאכלו אבותינו בארץ מצרים. כל הרעב – יבוא ויאכל, כל הצריך – יבוא ויחגוג פסח. השנה כאן, לשנה הבאה בארץ ישראל. השנה עבדים, לשנה הבאה בני חורין.

🎤 מה נשתנה אינטראקטיבי

לחצו על כל שאלה כדי לגלות אותה – הילד/ה הקטנ/ה שרה!

שאלה 1: מַה נִּשְׁתַּנָּה הַלַּיְלָה הַזֶּה מִכָּל הַלֵּילוֹת? שֶׁבְּכָל הַלֵּילוֹת אָנוּ אוֹכְלִין חָמֵץ וּמַצָּה – הַלַּיְלָה הַזֶּה כֻּלּוֹ מַצָּה. 👆
שאלה 2: שֶׁבְּכָל הַלֵּילוֹת אָנוּ אוֹכְלִין שְׁאָר יְרָקוֹת – הַלַּיְלָה הַזֶּה מָרוֹר. 👆
שאלה 3: שֶׁבְּכָל הַלֵּילוֹת אֵין אָנוּ מַטְבִּילִין אֲפִילוּ פַּעַם אֶחָת – הַלַּיְלָה הַזֶּה שְׁתֵּי פְעָמִים. 👆
שאלה 4: שֶׁבְּכָל הַלֵּילוֹת אָנוּ אוֹכְלִין בֵּין יוֹשְׁבִין וּבֵין מְסֻבִּין – הַלַּיְלָה הַזֶּה כֻּלָּנוּ מְסֻבִּין. 👆
עֲבָדִים הָיִינוּ לְפַרְעֹה בְּמִצְרָיִם,
וַיּוֹצִיאֵנוּ יְיָ אֱלֹהֵינוּ מִשָּׁם
בְּיָד חֲזָקָה וּבִזְרוֹעַ נְטוּיָה.
וְאִלּוּ לֹא הוֹצִיא הַקָּדוֹשׁ בָּרוּךְ הוּא
אֶת אֲבוֹתֵינוּ מִמִּצְרַיִם,
הֲרֵי אָנוּ וּבָנֵינוּ וּבְנֵי בָנֵינוּ
מְשֻׁעְבָּדִים הָיִינוּ לְפַרְעֹה בְּמִצְרָיִם.
