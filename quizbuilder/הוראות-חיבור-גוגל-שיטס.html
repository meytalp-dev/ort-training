הוראות הגדרה – חיבור ציונים ל-Google Sheets

פעם אחת בלבד! אחרי זה הכל אוטומטי

1

פתחו Google Sheet חדש

היכנסו ל-sheets.google.com וצרו גיליון חדש.

תנו לו שם, למשל: ציוני מבחן – מודל לינארי

2

פתחו את עורך הסקריפט

בתפריט למעלה לחצו על: תוספים (Extensions) ← Apps Script

ייפתח חלון חדש של עורך קוד.

3

מחקו את הקוד הקיים והדביקו את הקוד הזה:

סמנו את כל הקוד למטה, העתיקו (Ctrl+C), ובעורך הסקריפט מחקו הכל והדביקו (Ctrl+V):

function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); // Use quiz name as sheet tab, or default var sheetName = data.quiz || 'ציונים'; var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); } // Build row from data var keys = Object.keys(data); var fixedKeys = ['date','time','quiz','name','studentClass', 'multipleScore','totalScore','percentage']; // Add headers if sheet is empty if (sheet.getLastRow() === 0) { var headers = ['תאריך','שעה','שאלון','שם תלמיד','כיתה', 'אמריקאיות','ציון כולל','אחוז']; // Add dynamic question columns keys.forEach(function(k) { if (fixedKeys.indexOf(k) === -1) headers.push(k); }); sheet.appendRow(headers); var hRange = sheet.getRange(1, 1, 1, headers.length); hRange.setFontWeight('bold'); hRange.setBackground('#0891b2'); hRange.setFontColor('white'); } // Build row var row = [ data.date, data.time, data.quiz, data.name, data.studentClass, data.multipleScore, data.totalScore, data.percentage ]; keys.forEach(function(k) { if (fixedKeys.indexOf(k) === -1) row.push(data[k]); }); sheet.appendRow(row); return ContentService .createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput('The script is working!') .setMimeType(ContentService.MimeType.TEXT); }
4

שמרו ופרסמו

1. לחצו Ctrl+S לשמירה

2. לחצו על Deploy (פריסה) ← New deployment

3. ליד "Select type" לחצו על גלגל השיניים ובחרו Web app

4. הגדרות:

  • Description: quiz results
  • Execute as: Me
  • Who has access: Anyone

5. לחצו Deploy

6. אשרו את ההרשאות (Authorize access → בחרו את החשבון שלכם → Allow)

אם מופיע "Google hasn't verified this app" – לחצו Advanced ← Go to (unsafe). זה בסדר, כי זה הסקריפט שלכם.
5

העתיקו את הקישור

אחרי הפריסה יופיע Web app URL – זה קישור ארוך שמתחיל ב:

https://script.google.com/macros/s/XXXX.../exec

העתיקו את הקישור הזה והדביקו אותו במחולל השאלונים (בשלב "הגדרות" יש שדה לקישור).

זהו! מאותו רגע כל ציון ייכנס אוטומטית לטבלה שלכם ב-Google Sheets.
כל שאלון יקבל גיליון (tab) נפרד בטבלה!
